Starting a prototype

# Start all at once (Windows)
start-all.bat

# Or start one individually
cd ai-learning-dashboard\backend  &&  uvicorn main:app --reload --port 8000
cd ai-learning-dashboard\frontend &&  npm run dev

cd ai-prioritization-tool\backend  &&  uvicorn main:app --reload --port 8001
cd ai-prioritization-tool\frontend &&  npm run dev

cd mcp-explorer\backend  &&  uvicorn main:app --reload --port 8002
cd mcp-explorer\frontend &&  npm run dev
